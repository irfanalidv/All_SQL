  p.name,
  /*u.email, */
  c.name,
  date_trunc('day', c.created_at)::date as date

  inner join users u
  ON u.id = c.user_id
  inner join projects p
  ON c.project_id = p.id

  AND a.answered_at IS NOT NULL
